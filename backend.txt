# 🔥 NODE 3: GENERATE QUERY
def sql_node(state: AgentState):
    if state.get("error"):
        return state

    prompt = f"""
    Given the schema:
    {state['schema_sql']}

    Answer the question:
    {state['question']}

    Return ONLY SQL query.
    """

    sql = generate_sql(prompt)
    sql = clean_sql(sql)

    state["sql"] = sql
    return state


# 🔥 NODE 4: VALIDATE
def validate_node(state: AgentState):
    if state.get("error"):
        return state

    is_valid, error = validate_sql(state["sql"])

    if not is_valid:
        state["error"] = error

    return state


# 🔥 NODE 5: EXECUTE QUERY
def execute_node(state: AgentState):
    if state.get("error"):
        return state

    db = SessionLocal()
    try:
        result = execute_sql(state["sql"], db)
        state["result"] = result
    except Exception:
        state["error"] = "Query execution failed"
    finally:
        db.close()

    return state


# 🔥 NODE 6: CACHE
def cache_node(state: AgentState):
    if state.get("error"):
        return state

    key = generate_cache_key(state["question"])

    value = serialize_response({
        "sql": state["sql"],
        "result": state["result"]
    })

    set_cache(key, value)

    return state


# 🔥 BUILD GRAPH
def build_agent():
    graph = StateGraph(AgentState)

    # graph.add_node("schema", schema_node)
    # graph.add_node("schema_exec", schema_exec_node)
    graph.add_node("sql", sql_node)
    graph.add_node("validate", validate_node)
    graph.add_node("execute", execute_node)
    graph.add_node("cache", cache_node)

    graph.set_entry_point("schema")

    graph.add_edge("schema", "schema_exec")
    graph.add_edge("schema_exec", "sql")
    graph.add_edge("sql", "validate")
    graph.add_edge("validate", "execute")
    graph.add_edge("execute", "cache")

    return graph.compile()
